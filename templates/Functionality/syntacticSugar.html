array_sort()
Sorts the current array (in-place) 
												    function array_sort ( current_array , targetProperty ) {

												        // Sort objects - if targetProperty is provided 
												        if (targetProperty) {

												            current_array.sort(function(a, b) { return parseFloat(a[targetProperty]) - parseFloat(b[targetProperty]) })

												        // Sort numbers
												        } else if (!isNaN(current_array[0])) {

												            current_array.sort(function(a, b) { return parseFloat(a) - parseFloat(b) })

												        // Sort strings
												        } else {

												            current_array.sort()

												        }
												    };

// Argument Types :

[ current_array ] - must be Array

[ targetProperty ] - must be String (required only if sorting Array of Objects)

// Will result :

The current Array will be sorted in-place, in ascending order.

// Usage Example :

array_sort(myArrayOfObjects, 'objectPropToSortBy')

									    function array_shuffle ( current_array ) {

									        for (var i = current_array.length - 1; i > 0; i--) {
									            var j = Math.floor(Math.random() * (i + 1));
									            var temp = current_array[i];
									            current_array[i] = current_array[j];
									            current_array[j] = temp;
									        }

									    };

getParent()
Return Parent 
								    function getParent ( elem , identifier ) {

								        elemParent = elem.parentElement;

								        if (elemParent) {

								            var firstChar = identifier.charAt(0);

								            if   (verify()) { return elemParent                        } 
								            else            { return getParent(elemParent, identifier) }

								        } else { return null };
								        
 


								        // Check if elemParent has the required identifier
								        function verify() {

								            // remove first char - for class/id cases
								            var identif_substr = identifier.substring(1); 

								            // search for class
								            if (firstChar === '.') { 

								                return ( " " + elemParent.className + " " ).indexOf( " " + identif_substr + " " ) > -1 

								            // search for id
								            } else if (firstChar === '#') { 

								                return (elemParent.id == identif_substr) 

								            // search for tag
								            } else { 

								                return (elemParent.tagName == identifier.toUpperCase()) 

								            }
								        }
								    };

// Argument Types :

[ elem ] - must be HTMLElement

[ identifier ] - must be String

[ identifier ] - must follow the format : '.myClass' (dot prefix) | '#myId' (hastag prefix) | 'myTagName' (no prefix)

// Will return :

HTMLElement - the first parent which matches the query

null - if no parent matches the query

// Usage Example :

var targetParent = getParent(myHTMLElement, '.myClass');

getOwnedClasses()
Check Classes and Return all found 
								    function getOwnedClasses ( elem , ...classes ) {

								        // to accept both scenarios: array-of-classes || multiple class strings
								        if ( Array.isArray(classes[0]) ) { classes = classes[0] };

								        var resultsArray = classes.filter( klass => (" " + elem.className + " " ).indexOf( " " + klass + " " ) > -1 );

								        if (resultsArray.length > 0) { return resultsArray }
								        else                         { return null         }

								    };

// Argument Types :

[ elem ] - must be HTMLElement

[ classes ] - must be Strings or Array of Strings

// Will return :

Array of Strings - the classes that were found on the element

null - if no classes were found on the element

// Usage Example :

var ownedClasses = getOwnedClasses(myHTMLElement , 'myClass' , 'myOtherClass') ) (argument: strings)

var ownedClasses = getOwnedClasses(myHTMLElement, ['myClass', 'myOtherClass']) ) (argument: array of strings)

								    function hasAnyClass ( elem, ...classes ) {

								        // to accept both scenarios: array of class-strings || multiple individual class-strings

								    };
